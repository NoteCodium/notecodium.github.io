## Expensive pop

![image.png](/images/image-279.png)

Elements are inserted at b and f is from where pop should happen as this is a queue, so to remove 9 we have to start popping from front till we reach 9, pop there and finally pop one more time

insert in another tmp queue

![image.png](/images/image-280.png)

```
class MyStack {
private:
    queue<int> q;
public:
    MyStack(){}
    void push(int x) {
        q.push(x);
    }
    
    int pop() {
        if(empty()) return -1;
        int ans;
        queue<int> tmp;
        while(q.size()>1){
            tmp.push(q.front());
            q.pop();
        }
        ans=q.front();
        q.pop();
        q=tmp;
        return ans;
    }
    
    int top() {
        return q.back();
    }
    
    bool empty() {
        return q.empty();
    }
};

 */
```
