TreeNode* searchBST(TreeNode* root, int k) {
    TreeNode* node=root;
    while(node){
        if(node->val==k) return node;
        if(node->val<k) node=node->right;
        else node=node->left;
    }
    return NULL;
}

# getNodeAndParent

```cpp
pair<Node*, Node*> getNodeAndPar(Node* root, int k){
    Node* node=root;
    Node* par=nullptr;
    while(node){
        if(node->val==k) return {node,par};
        par=node;
        if(k>node->val) node=node->right;
        else node=node->left;
    }
    return {nullptr,nullptr};
}
```

pair<Node*, Node*> getNodeAndPar(Node* root, int k){

Node* maxNode(Node* root){
    Node* node=root;
    while(node->right){
        node=node->right;
    }
    return node;
}
