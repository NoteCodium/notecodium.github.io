https://leetcode.com/problems/kth-smallest-element-in-a-bst/description/

inorder traveral will give data in sorted manner, now doing it in without actually solving in the array

```
class Solution {
public:

    void t(TreeNode* node, int &ans, int &cnt, int k){
        if(!node or ans) return;
        t(node->left,ans,cnt,k);
        cnt++;        
        if(cnt==k) ans=node->val;
        t(node->right,ans,cnt,k);
    }

    int kthSmallest(TreeNode* root, int k) {
        //All the nodes are positive
        int ans=0,cnt=0;
        t(root,ans,cnt,k);
        return ans;
    }
};
```



https://leetcode.com/problems/range-sum-of-bst/?envType=daily-question&envId=2024-01-08

return the sum of values of all nodes with a value in the inclusive range [low, high].

```
class Solution {
public:
    int rangeSumBST(TreeNode* root, int l, int h) {
        if(!root) return 0;
        int val=root->val;
        if(val>=l and val<=h) return val+rangeSumBST(root->left,l,h)+rangeSumBST(root->right,l,h);
        if(val<l) return rangeSumBST(root->right,l,h);
        //val heen kam hain to umeeed karte hain uske right taraf bade wale range me aa jayenge
        else return rangeSumBST(root->left,l,h);
    }
};
```
