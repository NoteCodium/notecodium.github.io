ll fff(ll a, ll b, ll c){
    return (a-b)*(a-b) + (b-c)*(b-c) + (c-a)*(c-a);
}


void ffff(ll &ans, vl &l, vl &m, vl &h){
    for(ll x: m){
        vl tmp1,tmp2;   
        auto it1=lower_bound(it(l),x);
        if(it1!=l.end()){
            tmp1.push_back(*it1);
        }
        if(it1!=l.begin()){
            it1--;
            tmp1.push_back(*it1);
        }

        auto it2=lower_bound(it(h),x);
        if(it2!=h.end()){
            tmp2.push_back(*it2);
        }
        if(it2!=h.begin()){
            it2--;
            tmp2.push_back(*it2);
        }
        for(ll y: tmp1){
            for(ll z: tmp2){
                ans=min(ans,fff(x,y,z));
            }
        }
    }
}
