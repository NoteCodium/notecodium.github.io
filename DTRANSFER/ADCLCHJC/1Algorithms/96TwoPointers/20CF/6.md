```python
void solve(){
    ll n,c;
    inp(n,c);
    sinp(s);

    ll cntA=0,cntB=0;
    ll rudeness=0;
    ll ans=0;
    ll l=0,h=0;
    while(h<n){
        if(s[h]=='b'){
            cntB++;
        }
        if(s[h]=='a'){
            cntA++;
        }

        if(s[h]=='b'){
            rudeness+=cntA;
        }

        while(l<h and rudeness>c){
            if(s[l]=='a'){
                cntA--;
  
            }
            if(s[l]=='b'){
                cntB--;
            }

            if(s[l]=='a'){
              rudeness-=cntB;
            }

            l++;
        }
        if(rudeness<=c and l<=h){
            ans=max(ans,h-l+1);
        }
        h++;
    }
    print(ans);
    
}
```
