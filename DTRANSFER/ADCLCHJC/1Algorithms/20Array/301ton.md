https://leetcode.com/problems/find-all-duplicates-in-an-array/description/

 all the integers of nums are in the range [1, n]

each integer appears at most twice

return an array of all the integers that appears twice.

You must write an algorithm that runs in O(n) time and uses only constant auxiliary space, excluding the space needed to store the output

```python
class Solution {
    public:
        vector<int> findDuplicates(vector<int>& v) {
            int n=v.size();
            vector<int> ans;
            for(int i=0;i<n;i++){
                int x=abs(v[i]);
                int idx=x-1;
                if(v[idx]<0){
                    ans.push_back(idx+1);
                }
                else{
                    v[idx]*=-1;
                }
            }
            return ans;
        }
    };
```

```python
class Solution {
    public:
        int firstMissingPositive(vector<int>& v) {
            int n=v.size();
            bool contains1=0;
            for(int &x: v){
                if(x==1){
                    contains1=1;
                }
                if(x<=0 or x>n){
                    x=1;
                }
            }
    
            if(!contains1){
                return 1;
            }
    
            for(int i=0;i<n;i++){
                int x=abs(v[i]);
                int idx=x-1;
                if(v[idx]<0){
                    continue;
                }
                else{
                    v[idx]*=-1;
                }
            }
    
            for(int i=1;i<n;i++){
                if(v[i]>0){
                    return i+1;
                }
            }
    
            return n+1; 
        }
    };
```

```python
 vector<int> cyclicSort(vector<int> &v) {
    int n=v.size();
    for(int i=0;i<n;i++){
      while(v[i]!=n and v[i]!=i){
        int idx=v[i];
        swap(v[i],v[idx]);
      }
    }
    return v;
  }
```

```python
class Solution {
    public:
        int findKthPositive(vector<int>& v, int k) {
            int n=v.size();
            int num=1;
            for(int i=0;i<n and k>0;){
                if(v[i]==num){
                    i++;
                }
                else{
                    k--;
                }
                num++;
            }
            while(k--){
                num++;
            }
            return num-1;
            //imp
        }
    };
```

```python
class Solution {
    public:
        int findKthPositive(vector<int>& v, int k) {
            int n=v.size();
            int l=0,h=n-1;
    
            //if all elements 1 to n are present
            //then we have to find the k-th missing element
            //from n+1 to infinity, n+k
            int totalMissingElements=v[n-1]-n; //from 1 to n, how many numbers are missing
            if(k>totalMissingElements) return v[n-1]+(k-totalMissingElements);
    
            int totalMissingElementsBeforeArrayStart=v[0]-1;
            if(k<=totalMissingElementsBeforeArrayStart) return k;
            //kthMissingNumber<=totalMissingElements and kthMissingNumber>totalMissingElementsBeforeArrayStart
            int ansIdx;
            while(l<=h){
                int m=(l+h)/2;
                int missingTillM=v[m]-(m+1);    
                if(missingTillM<k){
                    ansIdx=m;
                    l=m+1;
                }
                else h=m-1;
            }
            int missingElementsTillAnsIdx=v[ansIdx]-(ansIdx+1);
            return v[ansIdx]+k-missingElementsTillAnsIdx;
        }
    };
```
